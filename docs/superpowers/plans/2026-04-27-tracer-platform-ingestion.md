# Tracer Platform + Ingestion Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up a new `tracer` monorepo with three Cloudflare Workers (ingest, consumer, api), Neon Postgres for project/key metadata, and ClickHouse Cloud for trace storage — traces flow end-to-end from HTTP POST to ClickHouse with API key auth.

**Architecture:** The ingest-worker validates a Bearer API key against CF KV (sub-millisecond), enqueues the OTLP payload to CF Queues, and immediately returns 202. The consumer-worker receives batched queue messages every ~5s and bulk-INSERTs spans into ClickHouse. The api-worker (Hono) manages projects and API keys via a REST API gated by an admin secret.

**Tech Stack:** TypeScript 5, Cloudflare Workers, CF KV, CF Queues, ClickHouse Cloud (HTTP API), Neon Postgres, Hono v4, Drizzle ORM, Zod, Vitest, pnpm workspaces, Wrangler 3

```typescript
export function generateRawKey(): string {
  const bytes = new Uint8Array(24)
  crypto.getRandomValues(bytes)
  const hex = Array.from(bytes)
    .map(b => b.toString(16).padStart(2, '0'))
    .join('')
  return `tr_live_${hex}`
}

export async function hashKey(raw: string): Promise<string> {
  const encoder = new TextEncoder()
  const data = encoder.encode(raw)
  const buffer = await crypto.subtle.digest('SHA-256', data)
  return Array.from(new Uint8Array(buffer))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('')
}

export function keyPrefix(raw: string): string {
  return raw.slice(0, 14)
}
```

```typescript
import type { MiddlewareHandler } from 'hono'

export function adminAuth(secret: string): MiddlewareHandler {
  return async (c, next) => {
    const incoming = c.req.header('X-Admin-Secret')
    if (!incoming || incoming !== secret) {
      return c.json({ error: 'unauthorized' }, 401)
    }
    await next()
  }
}
```

```typescript
import { Hono } from 'hono'
import { z } from 'zod'
import { projects } from '../../../../db/schema.ts'
import type { Db } from '../db.ts'
import { eq } from 'drizzle-orm'

const CreateProjectSchema = z.object({
  name: z.string().min(1),
  owner_email: z.string().email(),
})

export function projectsRouter(db: Db) {
  const router = new Hono()

  router.post('/', async c => {
    const result = CreateProjectSchema.safeParse(await c.req.json())
    if (!result.success) {
      return c.json({ error: 'invalid_payload', details: result.error.issues }, 400)
    }
    const [project] = await db
      .insert(projects)
      .values(result.data)
      .returning()
    return c.json(project, 201)
  })

  router.get('/:id', async c => {
    const project = await db.query.projects.findFirst({
      where: eq(projects.id, c.req.param('id')),
    })
```

```typescript
import { Hono } from 'hono'
import { z } from 'zod'
import { api_keys } from '../../../../db/schema.ts'
import type { Db } from '../db.ts'
import { generateRawKey, hashKey, keyPrefix } from '../keys.ts'
import { eq, and } from 'drizzle-orm'

const CreateKeySchema = z.object({
  name: z.string().optional(),
})

export function apiKeysRouter(db: Db, kv: KVNamespace) {
  const router = new Hono()

  router.post('/', async c => {
    const projectId = c.req.param('project_id')
    const result = CreateKeySchema.safeParse(await c.req.json().catch(() => ({})))
    if (!result.success) return c.json({ error: 'invalid_payload' }, 400)

    const raw = generateRawKey()
    const hash = await hashKey(raw)
    const prefix = keyPrefix(raw)

    const [key] = await db
      .insert(api_keys)
      .values({
        project_id: projectId,
        key_hash: hash,
        key_prefix: prefix,
        name: result.data.name ?? null,
      })
      .returning()

    await kv.put(hash, projectId)

    return c.json({
      id: key.id,
      raw_key: raw,
      key_prefix: key.key_prefix,
      name: key.name,
      created_at: key.created_at,
    }, 201)
  })

  router.get('/', async c => {
    const projectId = c.req.param('project_id')
    const keys = await db.query.api_keys.findMany({
      where: eq(api_keys.project_id, projectId),
    })
    return c.json(
      keys.map(({ key_hash: _omit, ...rest }) => rest)
    )
  })

  router.delete('/:kid', async c => {
    const projectId = c.req.param('project_id')
    const kid = c.req.param('kid')

    const key = await db.query.api_keys.findFirst({
      where: and(eq(api_keys.id, kid), eq(api_keys.project_id, projectId)),
    })
    if (!key) return c.json({ error: 'not_found' }, 404)

    await kv.delete(key.key_hash)
    await db
      .update(api_keys)
      .set({ revoked_at: new Date() })
      .where(eq(api_keys.id, kid))

    return c.json({ ok: true })
  })

  return router
}
```

```typescript
export function extractBearer(authHeader: string | null): string | null {
  if (!authHeader) return null
  if (!authHeader.startsWith('Bearer ')) return null
  const token = authHeader.slice('Bearer '.length).trim()
  return token.length > 0 ? token : null
}

export async function hashBearer(token: string): Promise<string> {
  const encoder = new TextEncoder()
  const data = encoder.encode(token)
  const buffer = await crypto.subtle.digest('SHA-256', data)
  return Array.from(new Uint8Array(buffer))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('')
}

/** Returns project_id if key is valid and active, null otherwise */
export async function lookupProjectId(
  authHeader: string | null,
  kv: KVNamespace
): Promise<string | null> {
  const token = extractBearer(authHeader)
  if (!token) return null
  const hash = await hashBearer(token)
  return kv.get(hash)
}
```

```typescript
import type { OtlpPayload } from '@tracer/types'

const OtlpAttributeValue = z.union([
  z.object({ stringValue: z.string() }),
  z.object({ intValue: z.number() }),
  z.object({ doubleValue: z.number() }),
  z.object({ boolValue: z.boolean() }),
])

  key: z.string(),
  value: OtlpAttributeValue,
})

const OtlpSpanSchema = z.object({
  traceId: z.string(),
  spanId: z.string(),
  parentSpanId: z.string().optional(),
  name: z.string(),
  kind: z.number().optional(),
  startTimeUnixNano: z.string(),
  endTimeUnixNano: z.string(),
  status: z.object({ code: z.number().optional(), message: z.string().optional() }).optional(),
  attributes: z.array(OtlpAttribute).optional(),
  events: z.array(z.object({
    timeUnixNano: z.string(),
    name: z.string(),
    attributes: z.array(OtlpAttribute).optional(),
  })).optional(),
})

const OtlpPayloadSchema = z.object({
  resourceSpans: z.array(z.object({
    resource: z.object({ attributes: z.array(OtlpAttribute).optional() }).optional(),
    scopeSpans: z.array(z.object({
      scope: z.object({ name: z.string().optional(), version: z.string().optional() }).optional(),
      spans: z.array(OtlpSpanSchema),
    })),
  })),
})

type ParseResult<T> = { ok: true; value: T } | { ok: false; issues: z.ZodIssue[] }

export function parseOtlpBody(raw: unknown): ParseResult<OtlpPayload> {
  const result = OtlpPayloadSchema.safeParse(raw)
  if (result.success) return { ok: true, value: result.data as OtlpPayload }
  return { ok: false, issues: result.error.issues }
}

export function countSpans(payload: OtlpPayload): number {
  return payload.resourceSpans.flatMap(rs => rs.scopeSpans).flatMap(ss => ss.spans).length
}
```

```typescript
import type { Env } from './env.ts'
import { lookupProjectId } from './auth.ts'
import { parseOtlpBody, countSpans } from './validate.ts'
import type { QueueMessage } from '@tracer/types'

export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    const url = new URL(request.url)

    if (request.method === 'GET' && url.pathname === '/health') {
      return Response.json({ ok: true, service: 'ingest-worker' })
    }

    if (request.method !== 'POST' || url.pathname !== '/v1/traces') {
      return Response.json({ error: 'not_found' }, { status: 404 })
    }

    const projectId = await lookupProjectId(
      request.headers.get('Authorization'),
      env.API_KEYS_KV
    )
    if (!projectId) {
      return Response.json({ error: 'unauthorized' }, { status: 401 })
    }

    let body: unknown
    try {
      body = await request.json()
    } catch {
      return Response.json({ error: 'invalid_payload', details: 'body must be JSON' }, { status: 400 })
    }

    const parsed = parseOtlpBody(body)
    if (!parsed.ok) {
      return Response.json({ error: 'invalid_payload', details: parsed.issues }, { status: 400 })
    }

    const spanCount = countSpans(parsed.value)
    if (spanCount > 500) {
      return Response.json({ error: 'payload_too_large', max: 500, received: spanCount }, { status: 413 })
    }

    const spans = parsed.value.resourceSpans.flatMap(rs => rs.scopeSpans).flatMap(ss => ss.spans)
    const message: QueueMessage = {
      project_id: projectId,
      spans,
      ingested_at: new Date().toISOString(),
    }

    try {
      await env.TRACES_QUEUE.send(message)
    } catch {
      return Response.json({ error: 'service_unavailable' }, { status: 503 })
    }

    return Response.json({ ok: true, span_count: spanCount }, { status: 202 })
  },
}
```

```typescript
import type { OtlpSpan, OtlpAttribute, ClickHouseRow } from '@tracer/types'

function flattenAttributes(attrs?: OtlpAttribute[]): Record<string, string> {
  if (!attrs) return {}
  const out: Record<string, string> = {}
  for (const attr of attrs) {
    const v = attr.value
    if ('stringValue' in v) out[attr.key] = v.stringValue
    else if ('intValue' in v) out[attr.key] = String(v.intValue)
    else if ('doubleValue' in v) out[attr.key] = String(v.doubleValue)
    else if ('boolValue' in v) out[attr.key] = String(v.boolValue)
  }
  return out
}

function detectKind(attrs: Record<string, string>): ClickHouseRow['kind'] {
  if (attrs['llm.model'] || attrs['gen_ai.system']) return 'llm'
  if (attrs['tool.name']) return 'tool'
  if (attrs['agent.step']) return 'agent'
  return 'chain'
}

function nanosToDateTime(nano: string): string {
  const ms = Number(BigInt(nano) / 1_000_000n)
  const d = new Date(ms)
  const pad = (n: number, z = 2) => String(n).padStart(z, '0')
  return (
    `${d.getUTCFullYear()}-${pad(d.getUTCMonth() + 1)}-${pad(d.getUTCDate())} ` +
    `${pad(d.getUTCHours())}:${pad(d.getUTCMinutes())}:${pad(d.getUTCSeconds())}.${pad(d.getUTCMilliseconds(), 3)}`
  )
}

function mapStatus(code?: number): ClickHouseRow['status'] {
  if (code === 1) return 'ok'
  if (code === 2) return 'error'
  return 'unset'
}

export function mapSpan(span: OtlpSpan, projectId: string, ingestedAt: string): ClickHouseRow {
  const attributes = flattenAttributes(span.attributes)
  return {
    project_id: projectId,
    trace_id: span.traceId,
    span_id: span.spanId,
    parent_span_id: span.parentSpanId ?? null,
    name: span.name,
    kind: detectKind(attributes),
    start_time: nanosToDateTime(span.startTimeUnixNano),
    end_time: nanosToDateTime(span.endTimeUnixNano),
    status: mapStatus(span.status?.code),
    attributes,
    events: (span.events ?? []).map(e => JSON.stringify(e)),
    ingested_at: ingestedAt,
  }
}
```

```typescript
import type { ClickHouseRow } from '@tracer/types'

export async function insertSpans(
  rows: ClickHouseRow[],
  clickhouseUrl: string,
  password: string
): Promise<void> {
  if (rows.length === 0) return

  const body = rows.map(r => JSON.stringify(r)).join('\n')
  const url = `${clickhouseUrl}/?query=${encodeURIComponent('INSERT INTO spans FORMAT JSONEachRow')}`

  const response = await fetch(url, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/x-ndjson',
      'X-ClickHouse-User': 'default',
      'X-ClickHouse-Key': password,
    },
    body,
  })

  if (!response.ok) {
    const text = await response.text()
    throw new Error(`ClickHouse insert failed: ${response.status} ${text}`)
  }
}
```

```typescript
export default {
  async queue(batch: MessageBatch<QueueMessage>, env: Env): Promise<void> {
    const rows = batch.messages.flatMap(msg =>
      msg.body.spans.map(span =>
        mapSpan(span, msg.body.project_id, msg.body.ingested_at)
      )
    )

    try {
      await insertSpans(rows, env.CLICKHOUSE_URL, env.CLICKHOUSE_PASSWORD)
      batch.ackAll()
    } catch (err) {
      console.error('ClickHouse insert failed, retrying batch:', err)
      batch.retryAll()
    }
  },
}
```
